13fn main() {
14    // 1. Initialize Serial Port
15    let port = serial_utils::get_serial_port(921_600);
16
17    // Clone for bidirectional communication
18    let mut reader = port.try_clone().expect("Failed to clone reader");
19    let mut writer = port;
20
21    // 2. Setup Shutdown Signal
22    // Arc allows both threads to own the same 'running' flag
23    let running = Arc::new(AtomicBool::new(true));
24    let r_running = Arc::clone(&running);
25
26    // 3. Telemetry Receiver Thread (STM32 -> PC)
27    let receiver_handle = thread::spawn(move || {
28        run_receiver(reader, r_running);
29    });
30    // 4. Main Sender Loop (PC -> STM32 @ 114Hz)
31    let mut csv_reader = csv::Reader::from_path("simulation_data.csv").expect("CSV missing");
32    let interval = Duration::from_secs_f32(1.0 / 114.0);
33
34    println!("Starting simulation...");
35    for result in csv_reader.deserialize() {
36        let start = Instant::now();
37        let row: VnSimulData = result.expect("CSV error");
38
39        // Send 130-byte packet
40        writer.write_all(&row.packet_to_vn()).expect("Write error");
41
42        let elapsed = start.elapsed();
43        if elapsed < interval {
44            thread::sleep(interval - elapsed);
45        }
46    }
47
48    // 5. Cleanup
49    println!("Simulation finished.");
50    running.store(false, Ordering::SeqCst); // Signal the receiver thread to stop
51    receiver_handle
52        .join()
53        .expect("Receiver thread failed to close cleanly");
54    println!("HIL system closed safely.");
55}

57pub fn process_stack(
58    receive_stack: &mut Vec<u8>,
59    wtr_vert: &mut csv::Writer<std::fs::File>,
60    wtr_horiz: &mut csv::Writer<std::fs::File>,
61    wtr_ctrl: &mut csv::Writer<std::fs::File>,
62) {
63    // We need at least 2 bytes (Sync + ID) to start processing
64    while receive_stack.len() >= 2 {
65        if receive_stack[0] == 0xAA {
66            // Sync byte
67            let packet_id = receive_stack[1];
68
69            // Calculate length dynamically based on structure size
70            // Note: We add 2 bytes for the header (Sync/ID)
71            let packet_len: usize = match packet_id {
72                0x01 => 2 + size_of::<VerticalFilterOutput>(),
73                0x02 => 2 + size_of::<HorizontalFilterOutput>(),
74                0x03 => 2 + size_of::<AirbakesControlerOutput>(),
75                _ => {
76                    // Unknown ID: remove corrupt sync byte and keep hunting
77                    receive_stack.remove(0);
78                    continue;
79                }
80            };
81
82            // Check if the full packet has arrived
83            if receive_stack.len() >= packet_len {
84                // Extract the exact number of bytes for this packet
85                let packet: Vec<u8> = receive_stack.drain(..packet_len).collect();
86
87                // Validate the VectorNav CRC
88                if protocol::is_packet_valid(&packet) {
89                    match packet_id {
90                        0x01 => {
91                            let data = protocol::parse_vertical(&packet);
92                            wtr_vert.serialize(data).unwrap();
93                        }
94                        0x02 => {
95                            let data = protocol::parse_horizontal(&packet);
96                            wtr_horiz.serialize(data).unwrap();
97                        }
98                        0x03 => {
99                            let data = protocol::parse_controller(&packet);
100                            wtr_ctrl.serialize(data).unwrap();
101                        }
102                        _ => {}
103                    }
104                }
105            } else {
106                // Full packet not yet received, wait for more data
107                break;
108            }
109        } else {
110            // Not a sync byte, remove it and keep looking
111            receive_stack.remove(0);
112        }
113    }
114}

116/// Dedicated function for the background receiver logic
117fn run_receiver(mut reader: Box<dyn SerialPort>, running: Arc<AtomicBool>) {
118    let mut receive_stack: Vec<u8> = Vec::new();
119    let mut serial_buf: Vec<u8> = vec![0; 1024];
120
121    let mut wtr_vert = csv::Writer::from_path("vertical_100hz.csv").unwrap();
122    let mut wtr_horiz = csv::Writer::from_path("horizontal_10hz.csv").unwrap();
123    let mut wtr_ctrl = csv::Writer::from_path("controller_10hz.csv").unwrap();
124
125    while running.load(Ordering::SeqCst) || !receive_stack.is_empty() {
126        if let Ok(n) = reader.read(serial_buf.as_mut_slice()) {
127            receive_stack.extend_from_slice(&serial_buf[..n]);
128        } else if !running.load(Ordering::SeqCst) {
129            break;
130        }
131
132        // Use the dynamic size_of matching logic here
133        process_stack(
134            &mut receive_stack,
135            &mut wtr_vert,
136            &mut wtr_horiz,
137            &mut wtr_ctrl,
138        );
139    }
140
141    // Final flush
142    // After the loop finishes, we force the buffers to disk
143    wtr_vert.flush().unwrap();
144    wtr_horiz.flush().unwrap();
145    wtr_ctrl.flush().unwrap();
146}
