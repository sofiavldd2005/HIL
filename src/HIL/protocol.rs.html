8/// deserialize to directly map from csv to struct : fields as written in the CSV
9///strut read from the csv and to be sent via serial port (excluding the headers and CRCs)
10///
11/// ```c char array : [8] to [127]```
12///### VectorNav IMU Structure Definition
13/// 
14/// This C structure represents the binary data layout for the IMU messages.
15///
16/// ```c
17/// typedef struct vn_output_msg_test {
18///     // sd card stuff
19///     uint8_t safe_bits[STRUCT_SAFE_BITS_LEN];
20///
21///     // char array : [0] a [129]
22///     // first 8 bytes are headers corresponding to the groups
23///     float yaw;                    // deg   [8]-[11]
24///     float pitch;                  // deg   [12]-[15]
25///     float roll;                   // deg   [16]-[19]
26///
27///     // m/s² (ou seja sem passar pelos filtros do vn)
28///     float uncomp_accel_x;         // [20]-[23]
29///     float uncomp_accel_y;         // [24]-[27]
30///     float uncomp_accel_z;         // [28]-[31]
31///
32///     // rad/s (ou seja sem passar pelos filtros do vn)
33///     float uncomp_angular_rate_x;  // [32]-[35]
34///     float uncomp_angular_rate_y;  // [36]-[39]
35///     float uncomp_angular_rate_z;  // [40]-[43]
36///
37///     float temp;                   // ºC    [44]-[47]
38///     float press;                  // kPa   [48]-[51]
39///
40///     float mag_x;                  // gauss [52]-[55]
41///     float mag_y;                  // gauss [56]-[59]
42///     float mag_z;                  // gauss [60]-[63]
43///
44///     // m/s² (ou seja, ja passando pelos filtros)
45///     float accel_x;                // [64]-[67]
46///     float accel_y;                // [68]-[71]
47///     float accel_z;                // [72]-[75]
48///
49///     // rad/s
50///     float angular_rate_x;         // [76]-[79]
51///     float angular_rate_y;         // [80]-[83]
52///     float angular_rate_z;         // [84]-[87]
53///
54///     float quat_x;                 // [88]-[91]
55///     float quat_y;                 // [92]-[95]
56///     float quat_z;                 // [96]-[99]
57///     float quat_w;                 // [100]-[103]
58///
59///     float lin_accel_x;            // [104]-[107]
60///     float lin_accel_y;            // [108]-[111]
61///     float lin_accel_z;            // [112]-[115]
62///
63///     float accel_N;                // [116]-[119]
64///     float accel_E;                // [120]-[123]
65///     float accel_D;                // [124]-[127]
66///
67///     // CRC [128]-[129]
68///
69///     // other stuff
70///     float accel_according_to_gravity;
71///     float altitude;
72///     uint32_t time;
73/// } vectornav_imu_typedef;
74/// ```
75///

116///### STM Output data struct
117/// 
118/// This C structure represents the binary data layout for the IMU messages.
119/// ```c
120/// Define Packet IDs
121/// 
122/// #define ID_VERTICAL   0x01  // 100Hz 
123/// #define ID_HORIZONTAL 0x02  // 10Hz
124/// #define ID_CONTROLLER 0x03  // 10Hz (or combined with horizontal)
125/// typedef struct __attribute__((packed)) {
126///     uint8_t sync_byte; // 0xAA
127///    uint8_t packet_id; // ID
128///     uint8_t payload[64]; // Size largest msg possible
129///     uint16_t crc;
130/// } OutgoingPacket_t;
131/// ```

134    pub altitude: f32,
135    pub velocity: f32,
136    pub acceleration: f32,

155/// we use this to convert the data from the structure into a X-byte VectorNav binary packet
156impl VnSimulData {
157    /// Converts the structure data into a 130-byte VectorNav binary packet.
158    ///
159    /// The first 8 bytes are headers (Sync, Group selection, etc.).
160    /// See the VN Control Center to verify these header bytes.
161    ///
162    /// # Return
163    /// Returns a `Vec<u8>` of length 130 ready for serial transmission.
164    ///
165    pub fn packet_to_vn(&self) -> Vec<u8> {
166        // to initialize the binary array to be sent via the serial port
167
168        let mut packet: Vec<u8> = vec![0; 130]; //initialize an array to zeros with the vec! macro
169        //Initalize headers -- see Vn Control center
170        packet[0] = 0xFA; //sync bit
171        packet[1] = 0x15;
172        packet[2] = 0x08;
173        packet[3] = 0x02;
174        packet[4] = 0x30;
175        packet[5] = 0x07;
176        packet[6] = 0xC4;
177        packet[7] = 0x00;
178
179        // Payload
180
181        // Euler Angles [8-19]
182        packet[8..12].copy_from_slice(&self.Yaw.to_le_bytes());
183        packet[12..16].copy_from_slice(&self.Pitch.to_le_bytes());
184        packet[16..20].copy_from_slice(&self.Roll.to_le_bytes());
185
186        // Uncomp Accel [20-31]
187        packet[20..24].copy_from_slice(&self.UncompAccX.to_le_bytes());
188        packet[24..28].copy_from_slice(&self.UncompAccY.to_le_bytes());
189        packet[28..32].copy_from_slice(&self.UncompAccZ.to_le_bytes());
190
191        // Uncomp Gyro [32-43]
192        packet[32..36].copy_from_slice(&self.UncompGyroX.to_le_bytes());
193        packet[36..40].copy_from_slice(&self.UncompGyroY.to_le_bytes());
194        packet[40..44].copy_from_slice(&self.UncompGyroZ.to_le_bytes());
195
196        // Temperature & Pressure [44-51]
197        packet[44..48].copy_from_slice(&self.Temperature.to_le_bytes());
198        packet[48..52].copy_from_slice(&self.Pressure.to_le_bytes());
199
200        // Mag [52-63]
201        packet[52..56].copy_from_slice(&self.MagX.to_le_bytes());
202        packet[56..60].copy_from_slice(&self.MagY.to_le_bytes());
203        packet[60..64].copy_from_slice(&self.MagZ.to_le_bytes());
204
205        // Accel [64-75]
206        packet[64..68].copy_from_slice(&self.AccelX.to_le_bytes());
207        packet[68..72].copy_from_slice(&self.AccelY.to_le_bytes());
208        packet[72..76].copy_from_slice(&self.AccelZ.to_le_bytes());
209
210        // Gyro [76-87]
211        packet[76..80].copy_from_slice(&self.GyroX.to_le_bytes());
212        packet[80..84].copy_from_slice(&self.GyroY.to_le_bytes());
213        packet[84..88].copy_from_slice(&self.GyroZ.to_le_bytes());
214
215        // Quat [88-103]
216        packet[88..92].copy_from_slice(&self.QuatX.to_le_bytes());
217        packet[92..96].copy_from_slice(&self.QuatY.to_le_bytes());
218        packet[96..100].copy_from_slice(&self.QuatZ.to_le_bytes());
219        packet[100..104].copy_from_slice(&self.QuatS.to_le_bytes());
220        // Lin Accel [104-115]
221        packet[104..108].copy_from_slice(&self.LinAccelX.to_le_bytes());
222        packet[108..112].copy_from_slice(&self.LinAccelY.to_le_bytes());
223        packet[112..116].copy_from_slice(&self.LinAccelZ.to_le_bytes());
224
225        // Accel NED [116-127]
226        packet[116..120].copy_from_slice(&self.LinAccelN.to_le_bytes());
227        packet[120..124].copy_from_slice(&self.LinAccelE.to_le_bytes());
228        packet[124..128].copy_from_slice(&self.LinAccelD.to_le_bytes());
229
230        //crc [128-129];
231        //let crc = calculate_crc(&p);
232        // packet[128..130].copy_from_slice(&crc.to_le_bytes());
233
234        packet[128] = 0xCC; // Dummy CRC
235        packet[129] = 0xAA;
236        packet
237    }
