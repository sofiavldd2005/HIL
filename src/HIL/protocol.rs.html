13/// ```c char array : [8] to [127]```
14///### VectorNav IMU Structure Definition
15/// 
16/// This C structure represents the binary data layout for the IMU messages.
17///
18/// ```c
19/// typedef struct vn_output_msg_test {
20///     // sd card stuff
21///     uint8_t safe_bits[STRUCT_SAFE_BITS_LEN];
22///
23///     // char array : [0] a [129]
24///     // first 8 bytes are headers corresponding to the groups
25///     float yaw;                    // deg   [8]-[11]
26///     float pitch;                  // deg   [12]-[15]
27///     float roll;                   // deg   [16]-[19]
28///
29///     // m/s² (ou seja sem passar pelos filtros do vn)
30///     float uncomp_accel_x;         // [20]-[23]
31///     float uncomp_accel_y;         // [24]-[27]
32///     float uncomp_accel_z;         // [28]-[31]
33///
34///     // rad/s (ou seja sem passar pelos filtros do vn)
35///     float uncomp_angular_rate_x;  // [32]-[35]
36///     float uncomp_angular_rate_y;  // [36]-[39]
37///     float uncomp_angular_rate_z;  // [40]-[43]
38///
39///     float temp;                   // ºC    [44]-[47]
40///     float press;                  // kPa   [48]-[51]
41///
42///     float mag_x;                  // gauss [52]-[55]
43///     float mag_y;                  // gauss [56]-[59]
44///     float mag_z;                  // gauss [60]-[63]
45///
46///     // m/s² (ou seja, ja passando pelos filtros)
47///     float accel_x;                // [64]-[67]
48///     float accel_y;                // [68]-[71]
49///     float accel_z;                // [72]-[75]
50///
51///     // rad/s
52///     float angular_rate_x;         // [76]-[79]
53///     float angular_rate_y;         // [80]-[83]
54///     float angular_rate_z;         // [84]-[87]
55///
56///     float quat_x;                 // [88]-[91]
57///     float quat_y;                 // [92]-[95]
58///     float quat_z;                 // [96]-[99]
59///     float quat_w;                 // [100]-[103]
60///
61///     float lin_accel_x;            // [104]-[107]
62///     float lin_accel_y;            // [108]-[111]
63///     float lin_accel_z;            // [112]-[115]
64///
65///     float accel_N;                // [116]-[119]
66///     float accel_E;                // [120]-[123]
67///     float accel_D;                // [124]-[127]
68///
69///     // CRC [128]-[129]
70///
71///     // other stuff
72///     float accel_according_to_gravity;
73///     float altitude;
74///     uint32_t time;
75/// } vectornav_imu_typedef;

158/// we use this to convert the data from the structure into a X-byte VectorNav binary packet
159impl VnSimulData {
160    /// Converts the structure data into a 130-byte VectorNav binary packet.
161    ///
162    /// The first 8 bytes are headers (Sync, Group selection, etc.).
163    /// See the VN Control Center to verify these header bytes.
164    ///
165    /// # Return
166    /// Returns a `Vec<u8>` of length 130 ready for serial transmission.
167    ///
168    pub fn packet_to_vn(&self) -> Vec<u8> {
169        // to initialize the binary array to be sent via the serial port
170
171        let mut packet: Vec<u8> = vec![0; 130]; //initialize an array to zeros with the vec! macro
172        //Initalize headers -- see Vn Control center
173        packet[0] = 0xFA; //sync bit
174        packet[1] = 0x15;
175        packet[2] = 0x08;
176        packet[3] = 0x02;
177        packet[4] = 0x30;
178        packet[5] = 0x07;
179        packet[6] = 0xC4;
180        packet[7] = 0x00;
181
182        // Payload
183
184        // Euler Angles [8-19]
185        packet[8..12].copy_from_slice(&self.Yaw.to_le_bytes());
186        packet[12..16].copy_from_slice(&self.Pitch.to_le_bytes());
187        packet[16..20].copy_from_slice(&self.Roll.to_le_bytes());
188
189        // Uncomp Accel [20-31]
190        packet[20..24].copy_from_slice(&self.UncompAccX.to_le_bytes());
191        packet[24..28].copy_from_slice(&self.UncompAccY.to_le_bytes());
192        packet[28..32].copy_from_slice(&self.UncompAccZ.to_le_bytes());
193
194        // Uncomp Gyro [32-43]
195        packet[32..36].copy_from_slice(&self.UncompGyroX.to_le_bytes());
196        packet[36..40].copy_from_slice(&self.UncompGyroY.to_le_bytes());
197        packet[40..44].copy_from_slice(&self.UncompGyroZ.to_le_bytes());
198
199        // Temperature & Pressure [44-51]
200        packet[44..48].copy_from_slice(&self.Temperature.to_le_bytes());
201        packet[48..52].copy_from_slice(&self.Pressure.to_le_bytes());
202
203        // Mag [52-63]
204        packet[52..56].copy_from_slice(&self.MagX.to_le_bytes());
205        packet[56..60].copy_from_slice(&self.MagY.to_le_bytes());
206        packet[60..64].copy_from_slice(&self.MagZ.to_le_bytes());
207
208        // Accel [64-75]
209        packet[64..68].copy_from_slice(&self.AccelX.to_le_bytes());
210        packet[68..72].copy_from_slice(&self.AccelY.to_le_bytes());
211        packet[72..76].copy_from_slice(&self.AccelZ.to_le_bytes());
212
213        // Gyro [76-87]
214        packet[76..80].copy_from_slice(&self.GyroX.to_le_bytes());
215        packet[80..84].copy_from_slice(&self.GyroY.to_le_bytes());
216        packet[84..88].copy_from_slice(&self.GyroZ.to_le_bytes());
217
218        // Quat [88-103]
219        packet[88..92].copy_from_slice(&self.QuatX.to_le_bytes());
220        packet[92..96].copy_from_slice(&self.QuatY.to_le_bytes());
221        packet[96..100].copy_from_slice(&self.QuatZ.to_le_bytes());
222        packet[100..104].copy_from_slice(&self.QuatS.to_le_bytes());
223        // Lin Accel [104-115]
224        packet[104..108].copy_from_slice(&self.LinAccelX.to_le_bytes());
225        packet[108..112].copy_from_slice(&self.LinAccelY.to_le_bytes());
226        packet[112..116].copy_from_slice(&self.LinAccelZ.to_le_bytes());
227
228        // Accel NED [116-127]
229        packet[116..120].copy_from_slice(&self.LinAccelN.to_le_bytes());
230        packet[120..124].copy_from_slice(&self.LinAccelE.to_le_bytes());
231        packet[124..128].copy_from_slice(&self.LinAccelD.to_le_bytes());
232
233        //crc [128-129];
234        //let crc = calculate_crc(&p);
235        // packet[128..130].copy_from_slice(&crc.to_le_bytes());
236        //packet[128] = 0xCC; // Dummy CRC
237        // packet[129] = 0xAA; 
238
239        // Calculate CRC using the VectorNav algorithm
240        let checksum = calculate_vn_crc(&packet);
241        let crc_bytes = checksum.to_le_bytes();
242        packet[128] = crc_bytes[0];
243        packet[129] = crc_bytes[1];
244        packet
245    }
246}

249/// Rust implementation of the VectorNav Datasheet CRC
250pub fn calculate_vn_crc(data: &[u8]) -> u16 {
251    let mut crc: u16 = 0;
252    // We start at index 1 to skip the sync byte '0xFA' as per your C code
253    for i in 1..128 {
254        let byte = data[i];
255        crc = ((crc >> 8) | (crc << 8)) & 0xFFFF;
256        crc ^= byte as u16;
257        crc ^= (crc & 0xFF) >> 4;
258        crc ^= crc << 12;
259        crc ^= (crc & 0xFF) << 5;
260    }
261    crc
262}

266///Checks if calculated CRC is equal to received CRC
267pub fn is_packet_valid(buffer: &[u8]) -> bool {
268    
269    // Ensure the buffer is at least long enough to hold a CRC (2 bytes)
270    if buffer.len() < 2 { return false; }
271
272    // 1. Extract the received CRC from the last two bytes of the buffer
273    let (data_part, crc_part) = buffer.split_at(buffer.len() - 2);
274    let received_crc = u16::from_le_bytes([crc_part[0], crc_part[1]]);
275
276    // 2. Calculate the CRC for the data portion
277    // Note: calculate_vn_crc must skip the sync byte internally
278    let calculated_crc = calculate_vn_crc(data_part);
279
280    calculated_crc == received_crc
281}

285pub fn parse_vertical(bytes: &[u8]) -> VerticalFilterOutput {
286    // Matches: safe_bits, altitude, velocity, acceleration, time
287    VerticalFilterOutput {
288        altitude: f32::from_le_bytes(bytes[PAYLOAD_OFFSET..PAYLOAD_OFFSET+4].try_into().unwrap()),
289        velocity: f32::from_le_bytes(bytes[PAYLOAD_OFFSET+4..PAYLOAD_OFFSET+8].try_into().unwrap()),
290        acceleration: f32::from_le_bytes(bytes[PAYLOAD_OFFSET+8..PAYLOAD_OFFSET+12].try_into().unwrap()),
291    }
292}

294pub fn parse_horizontal(bytes: &[u8]) -> HorizontalFilterOutput {
295    // Matches arrays: mag[3], pos[3], vel[3], acc[3], time
296    let mut o = PAYLOAD_OFFSET;
297    
298    let mut read_3f32 = |offset: &mut usize| {
299        let arr = [
300            f32::from_le_bytes(bytes[*offset..*offset+4].try_into().unwrap()),
301            f32::from_le_bytes(bytes[*offset+4..*offset+8].try_into().unwrap()),
302            f32::from_le_bytes(bytes[*offset+8..*offset+12].try_into().unwrap()),
303        ];
304        *offset += 12;
305        arr
306    };
307
308    HorizontalFilterOutput {
309        mag: read_3f32(&mut o),
310        pos: read_3f32(&mut o),
311        vel: read_3f32(&mut o),
312        acc: read_3f32(&mut o),
313    }
314}
